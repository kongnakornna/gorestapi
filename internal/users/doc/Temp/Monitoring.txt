## ✅ เพิ่ม Health & Monitoring Endpoints ให้กับ Server

จากความต้องการให้มี **Health Endpoints**, **Dependency Checks**, **Kubernetes Probes** และ **System Metrics** เราจะเพิ่มฟังก์ชันเหล่านี้ในเซิร์ฟเวอร์ Go ของคุณ โดยใช้โค้ดที่มีอยู่แล้ว (chi router) และเพิ่ม middleware สำหรับ monitoring

---

## 📁 ไฟล์ที่ต้องสร้าง/แก้ไข

1. **`internal/middleware/monitoring.go`** – (มีอยู่แล้ว แต่เราจะเพิ่มเติม)  
2. **`internal/server/health.go`** – สร้างใหม่ สำหรับ health handlers  
3. **`internal/server/handlers.go`** – เพิ่ม route `/health/*` และ `/metrics` (optional)

---

## 1. สร้าง `internal/server/health.go`

```go
package server

import (
	"encoding/json"
	"net/http"
	"runtime"
	"time"

	"github.com/go-chi/render"
	"github.com/redis/go-redis/v9"
	"gorm.io/gorm"
)

type HealthStatus struct {
	Status    string            `json:"status"`
	Timestamp time.Time         `json:"timestamp"`
	Checks    map[string]Check  `json:"checks,omitempty"`
}

type Check struct {
	Status  string `json:"status"`
	Message string `json:"message,omitempty"`
	Latency string `json:"latency,omitempty"`
}

// LivenessProbe - ตรวจสอบว่าเซิร์ฟเวอร์ทำงาน (ไม่ panic)
func LivenessProbe(w http.ResponseWriter, r *http.Request) {
	render.Respond(w, r, map[string]string{"status": "alive"})
}

// ReadinessProbe - ตรวจสอบว่าเซิร์ฟเวอร์พร้อมรับ traffic (DB, Redis พร้อม)
func ReadinessProbe(db *gorm.DB, redisClient *redis.Client) http.HandlerFunc {
	return func(w http.ResponseWriter, r *http.Request) {
		status := HealthStatus{
			Status:    "ok",
			Timestamp: time.Now(),
			Checks:    make(map[string]Check),
		}

		// Check Database
		startDB := time.Now()
		sqlDB, err := db.DB()
		if err != nil {
			status.Status = "unavailable"
			status.Checks["database"] = Check{Status: "fail", Message: err.Error()}
		} else {
			if err := sqlDB.Ping(); err != nil {
				status.Status = "unavailable"
				status.Checks["database"] = Check{Status: "fail", Message: err.Error()}
			} else {
				status.Checks["database"] = Check{Status: "ok", Latency: time.Since(startDB).String()}
			}
		}

		// Check Redis
		startRedis := time.Now()
		if redisClient != nil {
			if err := redisClient.Ping(r.Context()).Err(); err != nil {
				status.Status = "unavailable"
				status.Checks["redis"] = Check{Status: "fail", Message: err.Error()}
			} else {
				status.Checks["redis"] = Check{Status: "ok", Latency: time.Since(startRedis).String()}
			}
		} else {
			status.Checks["redis"] = Check{Status: "skip", Message: "redis client not configured"}
		}

		// Final status code
		code := http.StatusOK
		if status.Status != "ok" {
			code = http.StatusServiceUnavailable
		}
		render.Status(r, code)
		render.Respond(w, r, status)
	}
}

// DetailedHealth - รายละเอียดทั้งหมด (รวม system metrics)
func DetailedHealth(db *gorm.DB, redisClient *redis.Client) http.HandlerFunc {
	return func(w http.ResponseWriter, r *http.Request) {
		status := HealthStatus{
			Status:    "ok",
			Timestamp: time.Now(),
			Checks:    make(map[string]Check),
		}

		// DB
		startDB := time.Now()
		sqlDB, err := db.DB()
		if err != nil {
			status.Status = "degraded"
			status.Checks["database"] = Check{Status: "fail", Message: err.Error()}
		} else {
			if err := sqlDB.Ping(); err != nil {
				status.Status = "degraded"
				status.Checks["database"] = Check{Status: "fail", Message: err.Error()}
			} else {
				status.Checks["database"] = Check{Status: "ok", Latency: time.Since(startDB).String()}
			}
		}

		// Redis
		startRedis := time.Now()
		if redisClient != nil {
			if err := redisClient.Ping(r.Context()).Err(); err != nil {
				status.Status = "degraded"
				status.Checks["redis"] = Check{Status: "fail", Message: err.Error()}
			} else {
				status.Checks["redis"] = Check{Status: "ok", Latency: time.Since(startRedis).String()}
			}
		} else {
			status.Checks["redis"] = Check{Status: "skip", Message: "not configured"}
		}

		// System metrics (optional)
		var memStats runtime.MemStats
		runtime.ReadMemStats(&memStats)
		status.Checks["system"] = Check{
			Status: "ok",
			Message: "goroutines: " + itoa(runtime.NumGoroutine()) +
				", alloc_mb: " + itoa(memStats.Alloc/1024/1024) +
				", total_alloc_mb: " + itoa(memStats.TotalAlloc/1024/1024),
		}

		render.Respond(w, r, status)
	}
}

// BasicHealth - ง่ายๆ แค่สถานะ
func BasicHealth(w http.ResponseWriter, r *http.Request) {
	render.Respond(w, r, map[string]string{"status": "healthy"})
}

// Helper
func itoa(i int) string { return string(rune(i)) } // จริงควรใช้ strconv.Itoa แต่เราจะ import strconv
```

**แต่ควรใช้ `strconv.Itoa`** ดังนั้นให้แก้เป็น:

```go
import "strconv"
// แล้วใช้ strconv.Itoa(...)
```

## 2. เพิ่ม routes ใน `server/handlers.go`

ในฟังก์ชัน `New(...)` หลังจากที่สร้าง `apiRouter` ให้เพิ่ม:

```go
// Health endpoints
apiRouter.Get("/health", BasicHealth)
apiRouter.Get("/health/live", LivenessProbe)
apiRouter.Get("/health/ready", ReadinessProbe(db, redisClient))
apiRouter.Get("/health/detailed", DetailedHealth(db, redisClient))

// Optional metrics (ถ้าต้องการ)
apiRouter.Get("/metrics", middleware.MetricsHandler) // จาก middleware/monitoring.go
```

**หมายเหตุ:** `MetricsHandler` ต้องมีอยู่แล้วใน `internal/middleware/monitoring.go` (ตามที่คุณมีไฟล์ `monitoring.go`)

---

## 3. เปิดใช้งาน Monitoring Middleware (ถ้าต้องการ track request metrics)

ใน `server/handlers.go` ให้เพิ่ม middleware ก่อน routes:

```go
r.Use(middleware.MonitoringMiddleware) // หรือใส่ใน r.Use(...) ก่อน routes
```

---

## 4. ทดสอบ

```bash
# Basic health
curl http://localhost:8088/api/health

# Liveness probe
curl http://localhost:8088/api/health/live

# Readiness probe (เช็ค DB + Redis)
curl http://localhost:8088/api/health/ready

# Detailed health + metrics
curl http://localhost:8088/api/health/detailed

# Metrics snapshot
curl http://localhost:8088/api/metrics
```

---

## 📌 สรุป

| Endpoint | คำอธิบาย |
|----------|-----------|
| `GET /api/health` | basic status "healthy" |
| `GET /api/health/live` | liveness probe (always 200) |
| `GET /api/health/ready` | readiness probe (checks DB & Redis) |
| `GET /api/health/detailed` | detailed with DB/Redis latency, system metrics |
| `GET /api/metrics` | (optional) request metrics from monitoring middleware |

เพิ่มโค้ดตามนี้ แล้วเซิร์ฟเวอร์ของคุณจะพร้อมสำหรับ Kubernetes probes และ monitoring ทันทีครับ
